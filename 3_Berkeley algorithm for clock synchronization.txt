Server.java

import java.io.*;
import java.net.*;
import java.time.LocalDateTime;

public class Server {

    public static void main(String args[]) {

        try {

            ServerSocket ss = new ServerSocket(2000);

            System.out.println("Clock Server Started...\n");

            Socket s = ss.accept();

            System.out.println("Client Connected\n");

            DataInputStream dis =
                    new DataInputStream(s.getInputStream());

            DataOutputStream dos =
                    new DataOutputStream(s.getOutputStream());

            while(true) {

                // Receive client time
                String clientTime = dis.readUTF();

                System.out.println("Client Time Received : ");
                System.out.println(clientTime + "\n");

                // Server synchronized time
                String synchronizedTime =
                        LocalDateTime.now().toString();

                // Send synchronized time
                dos.writeUTF(synchronizedTime);

                System.out.println("Synchronized Time Sent : ");
                System.out.println(synchronizedTime + "\n");

                Thread.sleep(5000);
            }

        }
        catch(Exception e) {

            System.out.println(e);

        }
    }
}


Client.java

import java.io.*;
import java.net.*;
import java.time.LocalDateTime;

public class Client {

    public static void main(String args[]) {

        try {

            Socket s =
                    new Socket("localhost", 2000);

            DataInputStream dis =
                    new DataInputStream(s.getInputStream());

            DataOutputStream dos =
                    new DataOutputStream(s.getOutputStream());

            while(true) {

                // Client current time
                String clientTime =
                        LocalDateTime.now().toString();

                // Send time to server
                dos.writeUTF(clientTime);

                System.out.println("Client Time Sent : ");
                System.out.println(clientTime + "\n");

                // Receive synchronized time
                String synchronizedTime =
                        dis.readUTF();

                System.out.println(
                        "Synchronized Time From Server : ");

                System.out.println(
                        synchronizedTime + "\n");

                Thread.sleep(5000);
            }

        }
        catch(Exception e) {

            System.out.println(e);

        }
    }
}
